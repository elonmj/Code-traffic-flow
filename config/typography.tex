% Section style
\titleformat{\section}
  {\normalfont\Large\bfseries\color{darkblue}} % Format
  {\thesection}                % Label (e.g., "1.1")
  {0.75em}                     % Separation between label and title
  {}                           % Code before title (empty)
\titlespacing*{\section}{0pt}{3.5ex plus 1ex minus .2ex}{2.3ex plus .2ex} % Spacing

% Subsection style
\titleformat{\subsection}
  {\normalfont\large\bfseries\color{midblue}} % Format
  {\thesubsection}             % Label (e.g., "1.1.1")
  {0.75em}                     % Separation between label and title
  {}                           % Code before title (empty)
\titlespacing*{\subsection}{0pt}{3ex plus 0.8ex minus .2ex}{1.5ex plus .2ex} % Spacing (Adjusted slightly)

% --- List Styles ---
% Requires enumitem package (loaded in packages.tex)
% Requires xcolor package and color definition (darkblue)
\setlist[itemize]{leftmargin=1.5cm, label=\textcolor{darkblue}{$\blacktriangleright$}}
\setlist[enumerate]{leftmargin=1.5cm, label=\textcolor{darkblue}{\arabic*.}, font=\color{darkblue}\bfseries}
