% --- Header and Footer Design ---
% Requires fancyhdr package (loaded in packages.tex)
\pagestyle{fancy}
\fancyhf{} % Clear existing header/footer fields
\renewcommand{\headrulewidth}{0.4pt} % Rule below header
\renewcommand{\footrulewidth}{0.4pt} % Rule above footer

% Define content for header/footer (using benintraffic style)
% LE = Left Even, RO = Right Odd, RE = Right Even, LO = Left Odd
% \leftmark contains Chapter info, \rightmark contains Section info
\fancyhead[LE,RO]{\thepage} % Page number on outer side
\fancyhead[RE]{\textit{\nouppercase{\leftmark}}}  % Chapter title on inner side (even pages)
\fancyhead[LO]{\textit{\nouppercase{\rightmark}}} % Section title on inner side (odd pages)
% \fancyfoot[C]{\small Modélisation du Trafic Routier au Bénin} % Footer text - Replace with your thesis context if desired
\fancyfoot[C]{\small  Modélisation du Trafic Routier au Bénin} % Example footer
